### 第1步：Token无感刷新 ⭐⭐⭐⭐⭐

**现在的问题**：Token过期后，用户需要重新登录

**改进方案**：使用access_token + refresh_token

#### 1.1 修改request.js

```javascript
// src/utils/request.js
import axios from 'axios'
import { ElMessage } from 'element-plus'

const service = axios.create({
  baseURL: '/api',
  timeout: 5000
})

// 请求拦截器 - 添加token
service.interceptors.request.use(
  config => {
    const token = localStorage.getItem('access_token')
    if (token) {
      config.headers['Authorization'] = `Bearer ${token}`
    }
    return config
  },
  error => Promise.reject(error)
)

// 响应拦截器 - 处理token过期
let isRefreshing = false  // 防止多次刷新
let requests = []  // 缓存待处理的请求

service.interceptors.response.use(
  response => {
    const { data } = response
    if (data.code === '200') {
      return data.data
    } else if (data.code === '-1') {
      // Token过期
      if (!isRefreshing) {
        isRefreshing = true
        const refreshToken = localStorage.getItem('refresh_token')

        // 调用刷新token接口
        return service.post('/user/refresh-token', { refreshToken })
          .then(res => {
            // 保存新token
            localStorage.setItem('access_token', res.access_token)
            localStorage.setItem('refresh_token', res.refresh_token)

            // 重发缓存的请求
            requests.forEach(cb => cb(res.access_token))
            requests = []

            // 重发原请求
            return service(response.config)
          })
          .catch(() => {
            // 刷新失败，跳转登录
            localStorage.removeItem('access_token')
            localStorage.removeItem('refresh_token')
            window.location.href = '/auth/login'
            return Promise.reject('Token刷新失败')
          })
          .finally(() => {
            isRefreshing = false
          })
      } else {
        // 正在刷新，缓存请求
        return new Promise(resolve => {
          requests.push((token) => {
            response.config.headers['Authorization'] = `Bearer ${token}`
            resolve(service(response.config))
          })
        })
      }
    } else {
      ElMessage.error(data.msg || '请求失败')
      return Promise.reject(data.msg)
    }
  },
  error => Promise.reject(error)
)

export default service
```

### 第2步：动态菜单权限控制 ⭐⭐⭐⭐⭐

**现在的问题**：菜单是写死的，所有用户看到相同菜单

**改进方案**：根据权限动态生成菜单

#### 2.1 修改Pinia store

```javascript
// src/stores/admin.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useAdminStore = defineStore('admin', () => {
  const isCollapse = ref(false)
  const userInfo = ref(null)
  const routes = ref([])  // 用户可访问的路由
  const tabs = ref([])    // 打开的tab页签

  // 动态菜单 - 根据路由生成
  const menuList = computed(() => {
    return routes.value.filter(route => route.meta?.title)
  })

  // 设置用户信息和路由
  const setUserInfo = (info, userRoutes) => {
    userInfo.value = info
    routes.value = userRoutes
  }

  // Tab管理
  const addTab = (route) => {
    const exists = tabs.value.find(tab => tab.path === route.path)
    if (!exists) {
      tabs.value.push({
        title: route.meta?.title || '页面',
        path: route.path
      })
    }
  }

  const removeTab = (path) => {
    tabs.value = tabs.value.filter(tab => tab.path !== path)
  }

  const toggleCollapse = () => {
    isCollapse.value = !isCollapse.value
  }

  return {
    isCollapse,
    toggleCollapse,
    userInfo,
    routes,
    menuList,
    tabs,
    setUserInfo,
    addTab,
    removeTab
  }
})
```

#### 2.2 修改router/index.js

```javascript
// src/router/index.js
import { useAdminStore } from '@/stores/admin'

// 定义所有可能的路由
const allRoutes = [
  {
    path: '/back/dashboard',
    component: () => import('@/views/dashboard.vue'),
    meta: { title: '数据分析', icon: 'PieChart', roles: [1, 2] }
  },
  {
    path: '/back/knowledge',
    component: () => import('@/views/knowledge.vue'),
    meta: { title: '知识文章', icon: 'ChatLineSquare', roles: [2] }  // 只有管理员
  },
  {
    path: '/back/consultations',
    component: () => import('@/views/consultations.vue'),
    meta: { title: '咨询记录', icon: 'Message', roles: [2] }
  },
  {
    path: '/back/emotional',
    component: () => import('@/views/emotional.vue'),
    meta: { title: '情绪日志', icon: 'User', roles: [2] }
  }
]

// 根据权限过滤路由
const filterRoutes = (routes, userType) => {
  return routes.filter(route => {
    if (!route.meta?.roles) return true
    return route.meta.roles.includes(userType)
  })
}

// 路由守卫
router.beforeEach((to, from, next) => {
  const token = localStorage.getItem('access_token')
  const adminStore = useAdminStore()

  if (token) {
    const userInfo = JSON.parse(localStorage.getItem('userInfo'))

    // 第一次登录后，动态添加路由
    if (!adminStore.userInfo) {
      const accessibleRoutes = filterRoutes(allRoutes, userInfo.userType)
      adminStore.setUserInfo(userInfo, accessibleRoutes)

      // 动态添加路由
      accessibleRoutes.forEach(route => {
        router.addRoute('BackendLayout', route)
      })
    }

    // 权限检查
    if (to.path.startsWith('/back')) {
      const hasAccess = adminStore.routes.some(route => route.path === to.path)
      if (hasAccess) {
        next()
      } else {
        next('/back/dashboard')
      }
    } else {
      next()
    }
  } else {
    if (to.path.startsWith('/back')) {
      next('/auth/login')
    } else {
      next()
    }
  }
})
```

### 第3步：Tab页签 + 面包屑 ⭐⭐⭐⭐

**现在的问题**：没有tab页签，用户无法快速切换页面

**改进方案**：添加tab页签和面包屑导航

#### 3.1 创建TabBar组件

```vue
<!-- src/components/TabBar.vue -->
<template>
  <div class="tab-bar">
    <el-tabs
      v-model="activeTab"
      type="card"
      closable
      @tab-remove="removeTab"
      @tab-change="changeTab"
    >
      <el-tab-pane
        v-for="tab in adminStore.tabs"
        :key="tab.path"
        :label="tab.title"
        :name="tab.path"
      />
    </el-tabs>
  </div>
</template>

<script setup>
import { useAdminStore } from '@/stores/admin'
import { useRouter, useRoute } from 'vue-router'
import { computed } from 'vue'

const adminStore = useAdminStore()
const router = useRouter()
const route = useRoute()

const activeTab = computed({
  get: () => route.path,
  set: (path) => router.push(path)
})

const removeTab = (path) => {
  adminStore.removeTab(path)
  if (activeTab.value === path) {
    const remaining = adminStore.tabs
    if (remaining.length > 0) {
      router.push(remaining[remaining.length - 1].path)
    }
  }
}

const changeTab = (path) => {
  router.push(path)
}
</script>

<style scoped>
.tab-bar {
  padding: 10px 20px;
  background: #f5f7fa;
  border-bottom: 1px solid #dcdfe6;
}
</style>
```

### 第4步：数据可视化（ECharts） ⭐⭐⭐

**现在的问题**：后台没有数据图表

**改进方案**：使用ECharts添加数据可视化

#### 4.1 修改dashboard.vue

```vue
<!-- src/views/dashboard.vue -->
<template>
  <div class="dashboard">
    <div class="chart-container">
      <div ref="chartRef" style="width: 100%; height: 400px;"></div>
    </div>
  </div>
</template>

<script setup>
import { ref, onMounted, onUnmounted } from 'vue'
import * as echarts from 'echarts'
import { getAnalyticsOverview } from '@/api/admin'

const chartRef = ref(null)
let chart = null

const initChart = async () => {
  const data = await getAnalyticsOverview()

  const option = {
    title: { text: '用户情绪分布' },
    xAxis: {
      type: 'category',
      data: data.dates
    },
    yAxis: {
      type: 'value'
    },
    series: [
      {
        data: data.values,
        type: 'line',
        smooth: true
      }
    ]
  }

  chart = echarts.init(chartRef.value)
  chart.setOption(option)
}

// 响应式处理
const handleResize = () => {
  chart?.resize()
}

onMounted(() => {
  initChart()
  window.addEventListener('resize', handleResize)
})

onUnmounted(() => {
  window.removeEventListener('resize', handleResize)
  chart?.dispose()
})
</script>

<style scoped>
.dashboard {
  padding: 20px;
}

.chart-container {
  background: white;
  border-radius: 8px;
  padding: 20px;
  box-shadow: 0 2px 12px rgba(0, 0, 0, 0.1);
}
</style>
```

**优势**：
- ✅ 数据可视化
- ✅ 响应式图表
- ✅ 实时数据更新

**现在的问题**：刷新页面后，tab页签丢失

**改进方案**：使用localStorage缓存tab状态

#### 5.1 修改Pinia store

  const cached = localStorage.getItem('tabs')
  if (cached) {
    tabs.value = JSON.parse(cached)
  }
}

onMounted(() => {
  loadTabs()
})
